def average_of *indices
  begin # if an index is invalid then nget will cause an error
    ts = nget(*indices)
    return ts.sum(1)/ts.shape.last
  rescue
    return get(*indices).transpose.map {|xs| xs.inject(0,:+) / xs.length }
  end
end
